<body>
  <h1>💰 Expense Tracker</h1>
  <input id="desc" placeholder="Description" required>
  <input id="amt" type="number" placeholder="Amount (₹)" required>
  <button onclick="addExpense()">Add</button>
  
  <h3>Expenses:</h3>
  <ul id="list"></ul>
  <h3>Total: ₹<span id="total">0</span></h3>

  <script>
    let expenses = JSON.parse(localStorage.getItem("expenses")) || [];

    function updateList() {
      const list = document.getElementById("list");
      list.innerHTML = "";
      let total = 0;

      expenses.forEach((exp, i) => {
        total += +exp.amount;
        list.innerHTML += `<li>${exp.desc} - ₹${exp.amount} 
          <button onclick="delExpense(${i})">❌</button></li>`;
      });

      document.getElementById("total").textContent = total;
      localStorage.setItem("expenses", JSON.stringify(expenses));
    }

    function addExpense() {
      const desc = document.getElementById("desc").value.trim();
      const amt = document.getElementById("amt").value;

      if (!desc || !amt) return alert("Please fill all fields");

      expenses.push({ desc, amount: amt });
      document.getElementById("desc").value = "";
      document.getElementById("amt").value = "";
      updateList();
    }

    function delExpense(i) {
      expenses.splice(i, 1);
      updateList();
    }

    updateList();
  </script>
